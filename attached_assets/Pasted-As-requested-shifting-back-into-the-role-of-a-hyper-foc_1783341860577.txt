## 2. Post-Registration Token Claims & Routing Engine

A common architectural vulnerability is relying solely on client-side routing switches to serve dashboards. Veritas mitigates this by injecting **Custom User Claims** via Firebase Auth on user creation.

When a user hits the application shell, the Router reads the cryptographic JSON Web Token (JWT) state and dispatches the corresponding view layout.

### The Registration and Dispatch Lifecycle

```
[ Frontend Registration Form ]
              │
              ▼
[ Firebase Auth Create User ] ──(Triggers)──► [ Python Cloud Function: on_user_created ]
                                                        │
                                                        ▼
                                             1. Write User Doc to Firestore
                                             2. Inject Custom Claims (e.g., role: "educator")
                                                        │
                                                        ▼
[ Dashboard Dispatcher ] ◄──(Pushes JWT Claims)─────────┘
        │
        ├─► If role == "educator" ──► Mount: [ Educator Workspace ]
        ├─► If role == "admin"    ──► Mount: [ Institutional Admin Control Panel ]
        └─► If role == "auditor"   ──► Mount: [ Compliance & Audit Ledger Hub ]

```

## 3. Personalized Dashboard Variations (The Matrix)

### A. The Educator Dashboard: "The Workstation"

* **Primary Objective:** High-throughput document processing and manual grading verification.
* **Core UI Layout Components:** Interactive file-drop zone, live OCR split-view queue, current class ledger metrics, and export history hooks.
* **Data Scope:** Isolated exclusively to records where `userId == auth.uid`.

### B. The Institutional Administrator Dashboard: "The Command Center"

* **Primary Objective:** Global user provisioning, resource allocation telemetry, and organizational oversight.
* **Core UI Layout Components:** Departmental system usage graphs (API consumption metrics), educator account approval pipelines, and institutional global grade-distribution analytics.
* **Data Scope:** Read/Write clearance across the entire tenant organization group collection.

### C. The System Auditor Dashboard: "The Compliance Ledger"

* **Primary Objective:** Read-only immutable access for security, drift metrics, and processing transparency verification.
* **Core UI Layout Components:** System error logs, OCR algorithmic variance graphs (tracking human-corrected inputs vs. AI confidence metrics), and tamper-evident cryptographic hash verifications for generated Excel outputs.
* **Data Scope:** Globally distributed read-only stream access across systemic audit log collection nodes.

## 4. Production-Grade Implementation Script (`app.js`)

Append this logic to your unified application controller to support the registration mapping, claim checking, and dynamic dashboard provisioning.

```javascript
// Extended App State for Dynamic Multi-Tenancy Architecture
appState.userProfile = {
    role: null,         // "educator" | "admin" | "auditor"
    organization: "",
    preferences: {}
};

/**
 * Handles security registration and pushes profile metadata to the backend
 */
async function executeSecureRegistration(email, password, fullName, selectedRole) {
    displayGlobalLoadingState(true);
    try {
        // 1. Create native auth credentials
        const userCredential = await auth.createUserWithEmailAndPassword(email, password);
        const user = userCredential.user;

        // 2. Build personalized profile database payload
        const profilePayload = {
            uid: user.uid,
            name: fullName,
            email: email,
            role: selectedRole, // educator, admin, auditor
            created_at: firebase.firestore.FieldValue.serverTimestamp(),
            preferences: {
                theme: "glass-dark",
                defaultExportFormat: "xlsx"
            }
        };

        // 3. Persist profile document to global Firestore collection
        await db.collection('users').doc(user.uid).set(profilePayload);
        
        // Update runtime tracking state
        appState.currentUser = user;
        appState.userProfile = profilePayload;

        // 4. Dispatch the dynamically generated view layout based on role context
        evaluateRoleAndDispatchDashboard(selectedRole);

    } catch (error) {
        console.error("Critical Registration Pipeline Fault:", error);
        document.getElementById('authError').innerText = error.message;
    } finally {
        displayGlobalLoadingState(false);
    }
}

/**
 * Evaluates the authenticated user identity matrix and performs role-based mounting
 */
function evaluateRoleAndDispatchDashboard(role) {
    const mainWorkspace = document.querySelector('main');
    
    // Clear out existing dynamic views to avoid cross-tenant memory leakage
    const existingDynamicView = document.getElementById('dynamic-dashboard-context');
    if (existingDynamicView) existingDynamicView.remove();

    // Create a contextual dynamic shell layout container
    const dynamicShell = document.createElement('div');
    dynamicShell.id = 'dynamic-dashboard-context';
    dynamicShell.className = 'view-section active';

    // Tailor layout generation based completely on authenticated token roles
    switch (role) {
        case 'educator':
            dynamicShell.innerHTML = `
                <div class="dashboard-header">
                    <h1>Educator Grading Matrix</h1>
                    <p style="color: var(--text-muted);">Welcome back, ${appState.userProfile.name || 'Professor'}</p>
                </div>
                <div class="split-view">
                    <div class="glass-card"><h3>Active Class Ledgers</h3><div id="dashboard-content"></div></div>
                    <div class="glass-card"><h3>OCR Scanner Quick-Launch</h3><button class="btn" onclick="switchView('ocr-scanner')">Initialize Camera Stream</button></div>
                </div>`;
            mainWorkspace.appendChild(dynamicShell);
            initializeRealtimeSync(); // Standard pipeline execution hook
            break;

        case 'admin':
            dynamicShell.innerHTML = `
                <div class="dashboard-header">
                    <h1>Institutional Command Center</h1>
                    <p style="color: var(--text-muted);">System Domain Control Platform Node</p>
                </div>
                <div style="display:grid; grid-template-columns: repeat(3, 1fr); gap:1.5rem;">
                    <div class="glass-card"><h3>Total Active Faculty</h3><p style="font-size:2rem; font-weight:bold; color:var(--accent-blue);">42</p></div>
                    <div class="glass-card"><h3>OCR API Consumption</h3><p style="font-size:2rem; font-weight:bold; color:var(--accent-green);">89.4%</p></div>
                    <div class="glass-card"><h3>Pending System Tasks</h3><p style="font-size:2rem; font-weight:bold; color:#ef4444;">1</p></div>
                </div>`;
            mainWorkspace.appendChild(dynamicShell);
            break;

        case 'auditor':
            dynamicShell.innerHTML = `
                <div class="dashboard-header">
                    <h1>Compliance & Algorithmic Audit Hub</h1>
                    <p style="color: var(--text-muted);">Cryptographic Record Hash Verification Console</p>
                </div>
                <div class="glass-card" style="font-family: monospace; background:#000;">
                    <p style="color:#10b981;">[SYSTEM LOG LOGIC RUNTIME SEED: 2026-07-06T20:27:16Z]</p>
                    <p style="color:var(--text-muted);">SHA-256 Checksum Verification Match: Verified Ledger ID 01</p>
                    <p style="color:var(--text-muted);">OCR AI Confidence Shift Interval Baseline Delta: 0.041% variance</p>
                </div>`;
            mainWorkspace.appendChild(dynamicShell);
            break;

        default:
            console.error("Illegal system operation exception: Unassigned user role parameter initialized.");
            auth.signOut();
            window.location.reload();
    }
}

function displayGlobalLoadingState(isActive) {
    // Toggles layout skeletons/spinners contextually
    const loginButton = document.getElementById('btnLogin');
    if (loginButton) {
        loginButton.innerText = isActive ? "Authorizing Structural Identity Token..." : "Access Environment";
        loginButton.disabled = isActive;
    }
}
```
